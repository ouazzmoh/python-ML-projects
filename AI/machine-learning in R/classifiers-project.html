  #loop for folds
  mce_training_i <- c() #mean classification error for the iteration i
  mce_validation_i <- c() #mean classification error for the iteration i
  for (k in 1:K){
    valid_ik <- as.numeric(index_split[[k]]) #validation set
    pcaData_ik <- as.data.frame(pcaNAcont$x[,1:i])
    
    multinomModelPca_ik <- multinom(NAcont[-valid_ik,1]~. , data = pcaData_ik[-valid_ik,], trace=FALSE)
    
    prediction_valid_ik <- predict(multinomModelPca_ik, newdata=pcaData_ik[valid_ik,])
    prediction_training_ik <- predict(multinomModelPca_ik)
    
    conf_matrix_valid_ik <- table(prediction_valid_ik, NAcont[valid_ik,1])
    conf_matrix_training_ik <- table(prediction_training_ik, NAcont[-valid_ik,1])
    
    mce_training_i <- c(mce_training_i, 1 - sum(diag(conf_matrix_training_ik))/sum(conf_matrix_training_ik))
    mce_validation_i <- c(mce_validation_i, 1 - sum(diag(conf_matrix_valid_ik))/sum(conf_matrix_valid_ik))
  }
  
  mce_training <- c(mce_training, mean(mce_training_i))
  mce_validation <- c(mce_validation, mean(mce_validation_i))
  
  #Calculating confidence bounds for the classification error
  # Calculate the standard error of the mean
  se <- sd(mce_validation_i) / sqrt(length(mce_validation_i))
  
  # Calculate the t-value for a 95% confidence interval with n-1 degrees of freedom
  df <- length(mce_validation_i) - 1
  t_value <- qt(0.975, df)
  
  # Calculate the upper and lower bounds of the confidence interval
  upper_bounds <- c(upper_bounds, mean(mce_validation_i) + t_value * se)
  lower_bounds <- c(lower_bounds, mean(mce_validation_i) - t_value * se)
  
}

cross_validation_classification <- function(classification_model, from, to, by, predict_train=TRUE) {
  K <- 10
  vec <- c(1:nrow(NAcont))
  index_split <- split(sample(vec), 1:10)

  lower_bounds <- c()
  upper_bounds <- c()

  mce_training <- c()
  mce_validation <- c()
  
  # Loop for PCA's
  for (i in seq(from = from, to = to, by = by)){
    # Loop for folds
    mce_training_i <- c() # Mean classification error for the iteration i
    mce_validation_i <- c() # Mean classification error for the iteration i
    for (k in 1:K){
      valid_ik <- as.numeric(index_split[[k]]) # Validation set
      pcaData_ik <- as.data.frame(pcaNAcont$x[,1:i])
      
      multinomModelPca_ik <- classification_model(NAcont[-valid_ik,1] ~ . , data = pcaData_ik[-valid_ik,], trace=FALSE)
      
      prediction_valid_ik <- predict(multinomModelPca_ik, newdata=pcaData_ik[valid_ik,])
      if (predict_train){prediction_training_ik <- predict(multinomModelPca_ik, newdata=pcaData_ik[-valid_ik,])}
      
      conf_matrix_valid_ik <- table(prediction_valid_ik, NAcont[valid_ik,1])
      if (predict_train){conf_matrix_training_ik <- table(prediction_training_ik, NAcont[-valid_ik,1])}
      if (predict_train){ mce_training_i <- c(mce_training_i, 1 - sum(diag(conf_matrix_training_ik))/sum(conf_matrix_training_ik))}
      mce_validation_i <- c(mce_validation_i, 1 - sum(diag(conf_matrix_valid_ik))/sum(conf_matrix_valid_ik))
    }
    
    if(predict_train){mce_training <- c(mce_training, mean(mce_training_i))}
    mce_validation <- c(mce_validation, mean(mce_validation_i))
    
    # Calculating confidence bounds for the classification error
    # Calculate the standard error of the mean
    se <- sd(mce_validation_i) / sqrt(length(mce_validation_i))
    
    # Calculate the t-value for a 95% confidence interval with n-1 degrees of freedom
    df <- length(mce_validation_i) - 1
    t_value <- qt(0.975, df)
    
    # Calculate the upper and lower bounds of the confidence interval
    upper_bounds <- c(upper_bounds, mean(mce_validation_i) + t_value * se)
    lower_bounds <- c(lower_bounds, mean(mce_validation_i) - t_value * se)
  }
  
  # Return the metrics as a list
  metrics <- list(mce_training = mce_training,
                  mce_validation = mce_validation,
                  upper_bounds = upper_bounds,
                  lower_bounds = lower_bounds)
  return(metrics)
}
